\begin{prooftree}
  \AxiomC{\((e, s) \Downarrow v\)}
  \AxiomC{\((e', s) \Downarrow v'\)}
  \BinaryInfC{\((e \circ e', s) \Downarrow (v \circ v')\)}
\end{prooftree}

Where
\(\circ \in \{+_\N, -_\N, *_\N, /_\N, <_\N, >_\N, \leq_\N, \geq_\N,
=_{\N, \B, \textbf{Strings}}, \neq_{\N, \B, \textbf{Strings}},
\vee_\B, \wedge_\B, concat_{\textbf{Strings}}\}\).
